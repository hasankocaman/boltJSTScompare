Cypress Komutları

JS

JavaScript

describe('Cypress Commands', () => {
  it('should handle API requests', () => {
    cy.request('POST', '/api/login', {
      username: 'test@example.com',
      password: 'password'
    }).then((response) => {
      expect(response.status).to.eq(200)
      expect(response.body).to.have.property('token')
    })
  })
})

// Custom Commands
Cypress.Commands.add('loginViaUI', (username, password) => {
  cy.visit('/login')
  cy.get('[data-cy=username]').type(username)
  cy.get('[data-cy=password]').type(password)
  cy.get('[data-cy=login-btn]').click()
})
TS

TypeScript

// Tip güvenli API çağrısı
interface LoginResponse {
  token: string;
  user: { id: number; email: string; };
}

describe('Cypress Commands', (): void => {
  it('should handle API requests', (): void => {
    cy.request<LoginResponse>('POST', '/api/login', {
      username: 'test@example.com',
      password: 'password'
    }).then((response: Cypress.Response<LoginResponse>): void => {
      expect(response.status).to.eq(200)
      expect(response.body.user.email).to.be.a('string')
    })
  })
})

// Tip tanımlamalı Custom Commands

  namespace Cypress {
    interface Chainable {
      loginViaUI(username: string, password: string): Chainable<void>
    }
  }
}

    cy.get('.non-existent-element').click()

    if (error instanceof Error) {
      console.log('Element bulunamadı:', error.message)
    }
  }

// Custom hata sınıfı

  constructor(message: string, public testName: string) {
    super(message)

  }
}

Avantajlar ve Dezavantajlar

JavaScript

Avantajlar

  • Hızlı başlangıç
  • Esnek yapı
  • Daha az konfigürasyon
  • Geniş topluluk desteği
  • Runtime'da dinamik değişiklikler

Dezavantajlar

  • Runtime hataları
  • Tip güvenliği yok
  • IntelliSense desteği sınırlı
  • Refactoring zorluğu
  • Büyük projelerde bakım zorluğu

TypeScript

Avantajlar

  • Tip güvenliği
  • Compile-time hata yakalama
  • Mükemmel IntelliSense desteği
  • Kolay refactoring
  • Daha iyi kod dokumentasyonu
  • Büyük projelerde bakım kolaylığı

Dezavantajlar

  • Öğrenme eğrisi
  • Daha fazla konfigürasyon
  • Derleme süreci
  • Daha fazla kod yazma
  • Bazen aşırı karmaşık tip tanımları

Öneriler ve Sonuç

Genel Tavsiye

Cypress öğrenirken JavaScript ile başlayıp, deneyim kazandıkça TypeScript'e geçiş yapmak ideal bir yaklaşım olabilir.

JavaScript Kullanımı Önerilir:

  • Küçük ve orta ölçekli projeler
  • Hızlı prototipleme
  • Başlangıç seviyesi ekipler
  • Basit test senaryoları
  • Hızlı MVP geliştirme

TypeScript Kullanımı Önerilir:

  • Büyük ölçekli projeler
  • Kurumsal uygulamalar
  • Deneyimli geliştirici ekipleri
  • Kompleks test senaryoları
  • Uzun vadeli bakım gerektiren projeler

Sonuç

Her iki yaklaşımın da kendine özgü avantajları vardır. JavaScript basitlik ve hızlı geliştirme sunarken, TypeScript tip güvenliği ve uzun vadeli bakım kolaylığı sağlar.
